401  // Tache de gestion des entrées/sorties

403  costate

405  // Read Inputs
406  lireCommandes(&ec);
407 
408  // Si ARRÊT D'URGENCE

412  {
413 
414  /*if(ec.bp_video)

416 
417  }*/
418 
419  if(ec.bp_trim && !drone->fly)
420  initDrone(drone);
421 
422  if(ec.switch_land == 0 && !drone->fly) // Si on est en haut et que l'on ne vole pas
423  {
424  if( !(initDrone(drone) && takeoff(drone)) )
425  {
426  while(ec.switch_land == 0)
427  {
428  costate

431  yield;
432  }

443  }
444  }

449  {
450  while(ec.switch_land == 1)

471  // - haut et pas en vol
472  // - bas et en vol
473  // Ces cas sont interdit, et il n'y a rien à traiter dans ces cas.
474 
475 
476  // Si on est en vol

479  // Traiter la valeur des joystick

511  waitfor(DelayMs(5000));
512  }
513  else
514  {
515  ec.led_connecte = 0;

517  ec.led_debug = 1;
518  ecrireCommandes(ec);

520  connexion();
521 
522  ec.led_connecte = 1;
523  ec.led_erreur = 0;
524  ec.led_debug = 0;
525  ecrireCommandes(ec);
526  }
527  }
528  }

839 void connexion(void)
840 {
841  wifi_status status;
842  word waitms;
843  char connect_timeout; // pour la connexion
844 
845  connect = RECHERCHE;
846  sock_init();
847 
848  waitms = _SET_SHORT_TIMEOUT(300);
849 
850  while(connect != CONNECTE) // status.state != WLN_ST_ASSOC_ESS) // tant que la connexion n'est pas effective
851  {
852  printf("* Essai de connexion *\n");
853  connect_timeout = 1; // reset
854  tcp_tick(NULL);
855  ifdown(IF_WIFI0);
856  while(ifpending(IF_WIFI0) != IF_DOWN) // attend que l'éventuelle ancienne connexion soit arrêtée
857  tcp_tick(NULL);
858 
859  // Donne la fonction de callback avant la requête de scan
860  ifconfig(IF_WIFI0, IFS_WIFI_SCAN, scan_assoc_callback, IFS_END);
861  do
862  {
863  costate
864  {
865  tcp_tick(NULL);
866  waitfor(DelayMs(1));
867  }
868  costate
869  {
870  connect_timeout++; // jusqu'à l'overflow => sort de la boucle
871  waitfor(DelayMs(20)); // 5,10 secondes de tentatives de connexion
872  }
873  } while(connect != TROUVE && connect_timeout); // attend que le drone soit trouvé
874 
875 
876  if(connect == TROUVE) // (pas de timeout)
877  {
878  printf("Attente pour que la connexion soit effective\n");
879  connect_timeout = 1; // reset
880 
881  ifup(IF_WIFI0);
882  do
883  {
884  costate
885  {
886  ifconfig(IF_WIFI0, IFG_WIFI_STATUS, &status, IFS_END); // récupere le statut
887  while(!tcp_tick(NULL));
888  waitfor(DelayMs(1));
889  }
890  costate
891  {
892  connect_timeout++; // jusqu'à l'overflow => sort de la boucle
893  waitfor(DelayMs(20)); // 5,10 secondes de tentatives de connexion
894  }
895  } while(status.state != WLN_ST_ASSOC_ESS && connect_timeout); // attend que la connexion soit effective
896
